# 造数据

```python
# coding=utf8
import time
from elasticsearch import Elasticsearch
from elasticsearch import helpers
from ConnMysql_2 import ConnectMysql
es = Elasticsearch(hosts="localhost:9200")
connect = ConnectMysql('127.0.0.1', 'lucene', 'root', '123456', '3306')
def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        res = func(*args, **kwargs)
        print('共耗时约 {:.2f} 秒'.format(time.time() - start))
        return res
    return wrapper
@timer
def gen(start, offset):
    """ 使用生成器批量写入数据 """
    sql_cmd = "SELECT *  FROM lucene.tb_sku LIMIT %s,%s"
    res = connect.exec_query(sql_cmd, (start, offset), return_dict=True)
    action = [{
        "_index": "shop",
        "_type": "doc",
        "_source": item
    } for item in res]
    # print(action, type(action))
    helpers.bulk(es, action)
def get_total():
    sql_cmd = "SELECT count(id) as total FROM lucene.tb_sku;"
    return connect.exec_query(sql_cmd, return_dict=True)
if __name__ == '__main__':
    # total = get_total()
    gen(0, 1000000)
    print("执行完成")
```
