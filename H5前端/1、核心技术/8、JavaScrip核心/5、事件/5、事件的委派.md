<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <script>
        window.onload=function (ev) {
            /*
             *点击按钮以后添加超链接
             */
            var btn01=document.getElementById("btn01");
            btn01.onclick=function (ev1) {
                var li = document.createElement("li");
                li.innerHTML = "<a href=\"javascript: ;\" class=\"link\">我是一个超链接1</a>";
                //将li添加到ul中
                //alert(li.innerHTML);
                var u1 = document.getElementById("u1");
                u1.appendChild(li);
            }
                /*
                 *为每一个超链接都绑定单机响应函数
                 * 这里边我们为一个超链接都绑定一个单选响应函数，这种操作比较麻烦，
                 *  而且这些操作只能为已有的超链接设置事件，而新增的超链接需要重新绑定
                 */
                //获取所有的a
                var allA = document.getElementsByTagName("a");
                //遍历
            for(var i=0;i<allA.length;i++){
                allA[i].onclick=function (ev1) {
                    //alert("我是单机")
                }
            }
            /*
             *我们希望只绑定一次，即可应用到多个元素上，即是元素是后添加的
             * 我们可以尝试将其绑定给元素的共同的祖先元素
             *
             * 事件的委派
             *  -指的是事件统一绑定给共同的祖先元素，这样当后代元素上的事件触发时，会一直冒泡到祖先元素
             *      从而通过祖先元素的响应函数来处理事件
             *  -事件委派是利用了冒泡，通过委派可以减少事件绑定的次数，提高程序的性能
             */
            //
            document.getElementById("u1").onclick=function (ev1) {
                ev1=ev1||window.ev1;
                /*
                 *target
                 *  -ev1中的target表示触发事件的对象
                 */
                if(ev1.target.className=="link"){
                    alert(ev1.target);
                }
               

            }

        }
    </script>
</head>
<body>
    <button id="btn01">添加超链接</button>
    <ul id="u1">
        <li><a href="javascript: ;" class="link">我是一个超链接1</a> </li>
        <li><a href="javascript: ;" class="link">我是一个超链接2</a> </li>
        <li><a href="javascript: ;" class="link">我是一个超链接3</a> </li>

    </ul>

</body>
</html>
