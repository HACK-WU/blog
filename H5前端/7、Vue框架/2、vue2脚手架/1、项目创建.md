# 1、创建项目

```shell
npm install -g @vue/cli        #安装脚手架
vue create [vue_project]        #创建项目
然后进入交互模式
    1、选择vue版本以及打包工具
    2、选择包管理器npm/ymal
    
```

# 2、 脚手架文件结构

```
├── node_modules 
├── public
│   ├── favicon.ico:      页签图标
│   └── index.html:       主页面
├── src
│   ├── assets:           存放静态资源
│   │   └── logo.png
│   │── component:         存放组件
│   │   └── HelloWorld.vue
│   │── App.vue:           汇总所有组件
│   │── main.js:           入口文件
├── .gitignore:            git版本管制忽略的配置
├── babel.config.js:       babel的配置文件
├── package.json:          应用包配置文件 
├── README.md:             应用描述文件
├── package-lock.json：    包版本控制文件
```

# 3、详解main.js

```javascript
/* 
   该文件是整个项目的入口文件
*/
//引入Vue
//这里实际引入的是vue.runtime.xxx.js，是运行版的Vue，只包含：核心功能；没有模板解析器。
//所以在配置对象中不能使用template属性，否则报错。需要使用render属性；
import Vue from 'vue'     
//引入App组件，它是所有组件的父组件
import App from './App.vue'
//关闭vue的生产提示
Vue.config.productionTip = false
/* 
   关于不同版本的Vue：
   
      1.vue.js与vue.runtime.xxx.js的区别：
            (1).vue.js是完整版的Vue，包含：核心功能+模板解析器。
            (2).vue.runtime.xxx.js是运行版的Vue，只包含：核心功能；没有模板解析器。
      2.因为vue.runtime.xxx.js没有模板解析器，所以不能使用template配置项，需要使用
         render函数接收到的createElement函数去指定具体内容。
*/
//创建Vue实例对象---vm
new Vue({
   el:'#app',
   //render函数完成了这个功能：将App组件放入容器中
    render: h => h(App),
    //render 属性完整写法：
    render:function (createElement){    //++
        //这个匿名函数需要返回一个标签对象。这样页面才会显示对应的渲染内容。
        return createElement(App)  
        //运行时，脚手架会调用上面的匿名函数，并传入一个参数createElement
        //   -createElement是一个函数对象，主要用于创建标签元素，比如div,h1等等。
            有两种创建方式：
               1、指定标签名称，以及对应的内容：
                      createElement("h1","hello wordl") 
                        >> <h1> hello world</h1>    //这就是创建的结果
                2、指定vue文件，使用文件内template标签中指定内容进行创建
                      import App from "./App.vue"
                      createElement(App)     //这样就创建完成了。
    }
    
    //由于以上写法中并没有涉及到this的使用，所以可以使用箭头函数书写。
    render:(createElement)=>{    //++
        return createElement(App)
    }
    
    //然后箭头函数可以再次简写
    render:createEelement=> createEelment(App)    //++
    createEelement 太长了 ，简化为h
    >> render: h => h(App),
     
   // template:`<h1>你好啊</h1>`,
   // components:{App},
})
```
