def seek_book(booklist,name="null",id="null"):        #查找图书,可以按名称也可以按id
    if name != "null":
        for item in booklist:
            if item.get("name")==name:
                print(item)
                print("查询成功")
                return
    if id != "null":
        for item in booklist:
            if item.get("id")==id:
                print(item)
                print("查询成功")
                return
    print("查询失败")
