def insert_book(head,name,id,price):        #插入图书信息
    tmp=head.get("next")
    if tmp=="null":
        new_node={}
        new_node["id"]=id
        new_node["name"]=name
        new_node["price"]=price
        new_node["next"]="null"
        tmp=new_node
        head["next"]=tmp
        # print(head)
        return
    insert_book(tmp,name,id,price)
def show(head):                 #遍历链表
    tmp=head.get("next")
    if tmp=="null":
        # print("没有数据")
        return
    print(f"{tmp.get('name')} {tmp.get('id')} {tmp.get('price')}")
    show(tmp)
if __name__ == '__main__':
    head={}         #头结点
    head["name"]="单链表"
    head["length"]=0    #链表的长度
    head["next"]="null"
    insert_book(head,id=1,name="数据结构",price=20)
    insert_book(head,id=2,name="JSP",price=30)
    insert_book(head,id=3,name="ASP",price=40)
    # print(head)
    show(head)
