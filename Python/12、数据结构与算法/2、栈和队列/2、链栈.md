def push(head,name,id):         #进栈
    tmp=head.get("next")
    if tmp=="null":
        tmp={}
        tmp["name"]=name
        tmp["id"]=id
        tmp["next"]="null"
        head["next"]=tmp
        return
    push(tmp,name,id)
def pop(head,res="null"):                  #出栈
    tmp=head.get("next")
    if tmp=="null":
        return res
    if tmp.get("next")=="null":
        head["next"]="null"
        res=tmp
        return res
    return pop(tmp)

    push(head,"ASP",3)
