def push(head,name,id):         #入队
    tmp=head.get("next")
    if tmp=="null":
        tmp={}
        tmp["name"]=name
        tmp["id"]=id
        tmp["next"]="null"
        head["next"]=tmp
        return
    push(tmp,name,id)
def pop(head):                  #出队,并返回删除的元素
    tmp=head.get("next")
    if tmp=="null":
        return
    new_node=tmp.get("next")
    head["next"]=new_node
    tmp["next"]="null"
    return tmp
