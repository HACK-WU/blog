# 方法以二

```python
import os
import sys
#获取到当前文件的绝对路径
current_file_path=os.path.realpath(__file__)
#获取当前文件的所在目录的的绝对路径
dir=os.path.dirname(current_file_path)
root_path = os.path.dirname(dir)
#sys.path.append(root_path)
print(root_path)
```

# 方法三

```
import os,sys
import  inspect
#获取到当前模块的路径和文件信息
module_file=inspect.getfile(inspect.currentframe())
#获取到这个文件的绝对路径信息
module_path=os.path.abspath(module_file)
#根据需求获取到这个路径父目录
root_dir=os.path.dirname(module_path)
```
