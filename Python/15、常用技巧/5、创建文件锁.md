def _create_lock_file(lock_file):
    uuid = "0f81a077b294d3aa39a79a247fa6af"
    exists = False
    while Path(lock_file).exists():
        with open(lock_file, mode="rt", encoding="utf-8") as f:
            new_uuid = f.readline()
            if new_uuid.strip() == uuid:
                exists = True
                break
            else:
                exists = False
                lock_file += "1"
    if not exists:
        lock_file = Path(lock_file)
        lock_file.touch(exist_ok=False)
        if int(os.system(f"chattr +a {lock_file}")) != 0:
            lock_file.unlink(missing_ok=True)
            raise TryAgain("Failed to Add 'a' attribute")
        if int(os.system(f"echo {uuid} >> {lock_file}")) != 0:
            delete_lock_file(lock_file)
            raise TryAgain("Failed to write UUID.")
    dic = {
        "file_path": str(lock_file),
        "exists": exists
    }
    return dic
def _write_unique_tag(file_path, unique_tag):
    cmd = f"echo {unique_tag} >> {file_path}"
    if int(os.system(cmd)) != 0:
        raise TryAgain("Failed to write unique_tag.")
def delete_lock_file(file_path):
    """Delete lock file.
    :param file_path:
    :return:
    """
    num = int(os.popen(f"cat {file_path}| wc -l").read())
    os.system(f"chattr -a {file_path}")
    if num <= 2:
        Path(file_path).unlink(missing_ok=True)
    else:
        pid = os.getpid()
        if int(os.system(f"sed -i '/{pid}/ d' {file_path}")):
            raise TryAgain("Failed to delete lock file.")
        if int(os.system(f"chattr +a {file_path}")) != 0:
            raise TryAgain("Failed to Add 'a' attribute")
