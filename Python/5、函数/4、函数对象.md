# 利用函数对象，实现菜单功能

```
def fun1():
    print("功能1")
def fun2():
    print("功能2")
def fun3():
    print("功能3")
def quite():
    print("退出")
menu={
    0:["退出",quite],
    1:["功能1",fun1],
    2:["功能1",fun2],
    3:["功能1",fun3],
}
while True:
    for item in menu:
        print(item,menu[item][0])
    print("*"*30)
    option=input("请选择:\t").strip()
    if not option.isdigit():
        print("请输入功能对应的数字")
        continue
    option=int(option)
    if option not in menu:
        print("没有该选项")
        continue
    elif option==0:
        menu[option][1]()
        break
    else:
        menu[option][1]()   #调用函数
```
