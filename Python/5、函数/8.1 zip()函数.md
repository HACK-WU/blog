fruits = ['apple', 'banana', 'orange']
colors = ['red', 'yellow', 'pear']
dic={
    "1":"a",
    "2":"b",
    "3":"c"
}
quantities = [5, 2]
# 使用 zip() 打包多个可迭代对象
zipped = zip(fruits, colors,dic,quantities)
print(zipped)
# 遍历 zip() 返回的迭代器
for item in zipped:
    print(item)
    
##

('banana', 'yellow', '2', 2)
