from pythonds.basic.stack import Stack
def parenthesis_matching(string):
    """括号匹配，如果括号是左右对称的，就返回TRUE，否则返回False.
    """
    stack = Stack()
    for item in list(string):
        if item == "(":
            stack.push(item)
        if item == ")":
            # 当栈为空时，说明右括号没有对应的左括号，返回False
            if stack.isEmpty():
                return False
            stack.pop()
    return stack.isEmpty()
print(parenthesis_matching('((()))'))
print(parenthesis_matching('())'))
#
True
False
