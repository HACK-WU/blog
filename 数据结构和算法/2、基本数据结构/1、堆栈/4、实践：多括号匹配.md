from pythonds.basic.stack import Stack
def symbol_matching(string):
    """多括号匹配，如果括号是左右对称的，就返回TRUE，否则返回False."""
    stack = Stack()
    left = "([{"
    right = ")]}"
    for item in list(string):
        if len(item.strip()) == 0:
            continue
        if item in left:
            stack.push(item)
            continue
        index = right.find(item)
        if index == -1 or stack.isEmpty():
            return False
        if stack.peek() == left[index]:
            stack.pop()
            continue
        return False
    return stack.isEmpty()
#
print(symbol_matching('((( )))'))
print(symbol_matching('())'))
print(symbol_matching('{{([][])}()}'))
print(symbol_matching('[{()]'))
