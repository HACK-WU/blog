class SearchBinaryTree(object):
    def __init__(self):
        self._root = None
    def search(self, value):
        if self._root is None:
            return None, None
        pre = None
        node = self._root
        while node:
            if value == node.data:
                return pre, node
            pre = node
            if value < node.data:
                node = node.left
            else:
                node = node.right
        return pre, None
    def search_recursive(self, value):
        """递归查询"""
        pre = None
        def __search(node):
            nonlocal pre
            if node is None or node.data == value:
                return pre, node
            pre = node
            if value < node.data:
                return __search(node.left)
            return __search(node.right)
        return __search(self._root)
    def insert(self, value):
        if self._root is None:
            self._root = TreeNode(value)
            return
        pre, node = self.search(value)
        if node is not None:
            return
        if value < pre.data:
            pre.left = TreeNode(value)
        else:
            pre.right = TreeNode(value)
        return
    def remove(self, num: int):
        """删除节点"""
        # 若树为空，直接提前返回
        if self._root is None:
            return
        pre, cur = self.search(num)
        # 若无待删除节点，则直接返回
        if cur is None:
            return
        # 子节点数量 = 0 or 1
        if cur.left is None or cur.right is None:
            # 当子节点数量 = 0 / 1 时， child = null / 该子节点
            child = cur.left or cur.right
            # 删除节点 cur
            if cur != self._root:
                if pre.left == cur:
                    pre.left = child
                else:
                    pre.right = child
            else:
                # 若删除节点为根节点，则重新指定根节点
                self._root = child
        # 子节点数量 = 2
        else:
            # 获取中序遍历中 cur 的下一个节点
            tmp: TreeNode = cur.right
            while tmp.left is not None:
                tmp = tmp.left
            # 递归删除节点 tmp
            self.remove(tmp.data)
            # 用 tmp 覆盖 cur
            cur.data = tmp.data
    def get_root(self):
        return self._root
